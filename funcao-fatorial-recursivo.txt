jal main

fatorial:
	addi sp, sp, -8
	sw s0, 0(sp)
	sw ra, -4(sp)

	addi t0, zero, 1
	beq a0, t0, fatorial_fim
	beq a0, zero, fatorial_fim_zero
	addi s0, a0, 0
	addi a0, a0, -1
	jal ra, fatorial
	mul a0, s0, a0
	jal zero, fatorial_fim
	
	fatorial_fim_zero:
		addi a0, zero, 1
	fatorial_fim:
		addi sp, sp, 8
		lw s0, 0(sp) 
		lw ra, -4(sp)
		jalr zero, ra, 0

main:
	addi a0, zero, 5
	jal ra, fatorial
	
	addi a7, zero, 1
	ecall
